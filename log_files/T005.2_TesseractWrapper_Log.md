```python
def __init__(self, lang='spa+eng', config='--oem 3 --psm 6'):
    self.lang = lang
    self.config = config
    # Verify languages are available
    if verify_languages:
        available = pytesseract.get_languages()
        for l in lang.split('+'):
            if l not in available:
                raise RuntimeError(f"Language {l} not available")
```

```python
def extract_text(self, image) -> str:
    """Extract plain text from image."""
    return pytesseract.image_to_string(
        image,
        lang=self.lang,
        config=self.config
    )
```
